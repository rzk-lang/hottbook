```rzk
#define compose
  (A B C : U)
  (g : B → C)
  (f : A → B)
  : A → C
  := \ x → g (f x)

-- verify associativity
#def compose_assoc
  (A B C D : U)
  (f : A -> B)
  (g : B -> C)
  (h : C -> D)
  : compose A C D h (compose A B C g f) = compose A B D (compose B C D h g) f
  := refl
```
